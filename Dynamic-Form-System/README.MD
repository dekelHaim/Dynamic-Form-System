HOW TO RUN
Development (Direct)
bash
# Terminal 1: Start FastAPI
python main.py api

# Terminal 2: Open browser
http://localhost:8000


Production (Docker)
bash
# Start everything
docker-compose up -d

# Check status
docker-compose ps

# View logs
docker-compose logs -f api

# Stop everything
docker-compose down
TEST THE SYSTEM
1. Health Check
bash
curl http://localhost/health
# Response: {"status": "ok", "service": "..."}
2. Create Form
bash
curl -X POST http://localhost/api/forms/ \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Contact Form",
    "description": "Test form",
    "schema_definition": {
      "name": {"type": "string", "required": true},
      "email": {"type": "email", "required": true}
    }
  }'
3. Get Forms (Cached)
bash
curl http://localhost/api/forms/
# First call: ~100ms (DB)
# Second call: ~5ms (Cache)
4. Submit Data
bash
curl -X POST http://localhost/api/submissions/?form_id=1 \
  -H "Content-Type: application/json" \
  -d '{
    "data": {
      "name": "John",
      "email": "john@example.com"
    }
  }'
5. Bad Data (Rejected)
bash
curl -X POST http://localhost/api/submissions/?form_id=1 \
  -H "Content-Type: application/json" \
  -d '{
    "data": {
      "name": "ab",  # Too short!
      "email": "invalid"  # Not email!
    }
  }'
# Response: 422 error with validation details
